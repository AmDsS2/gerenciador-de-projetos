            card.addEventListener('click', function() {
                openModal('taskDetailsModal');
            });
        });
        
        // Simular clique no botão de novo projeto
        document.querySelector('button:contains("Novo Projeto")').addEventListener('click', function(e) {
            e.stopPropagation();
            openModal('newProjectModal');
        });
        
        // Atualizar progresso quando checkboxes são marcados
        document.querySelectorAll('input[type="checkbox"]').forEach(checkbox => {
            checkbox.addEventListener('change', function() {
                const checkedCount = document.querySelectorAll('input[type="checkbox"]:checked').length;
                const totalCount = document.querySelectorAll('input[type="checkbox"]').length;
                const progress = Math.round((checkedCount / totalCount) * 100);
                
                // Atualizar barra de progresso
                const progressBar = document.querySelector('.progress-bar');
                if (progressBar) {
                    progressBar.style.width = `${progress}%`;
                }
            });
        });
    </script>
